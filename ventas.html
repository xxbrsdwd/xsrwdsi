/* ===================================================== Módulo de Ventas - Sistema Brayan Raudales (orden de columnas fijo) Compatible con GitHub Pages (local + auto-commit opcional) ===================================================== */ let INVENTARIO = []; let VENTAS_EMPRESA = []; let ENVIOS_MOTO = []; let ENVIOS_CAEX = []; let VENTAS_RAUDA = []; // === Configuración GitHub (reutiliza la del inventario) === function loadGhCfg() { try { const s = localStorage.getItem("GH_SYNC_CFG"); return s ? JSON.parse(s) : {}; } catch { return {}; } } const GH_CFG = loadGhCfg(); // === Utilidades === function showToast(msg, type="success", ms=1500) { const t = document.getElementById("toast"); t.textContent = msg; return f.toLocaleDateString("es-HN")+" "+f.toLocaleTimeString("es-HN",{hour:"2-digit",minute:"2-digit"}); } async function fetchJSON(url){ try{const r=await fetch(url,{cache:"no-store"}); if(r.ok) return r.json();}catch{} return []; } function saveLocal(k,v){ localStorage.setItem(k, JSON.stringify(v)); } function loadLocal(k){ try{ const s=localStorage.getItem(k); return s?JSON.parse(s):[]; }catch{ return []; } } function b64(str){ return btoa(unescape(encodeURIComponent(str))); } // === Auto-commit GitHub === async function ghPutFile(path,content,msg="Auto-commit ventas"){ if(!GH_CFG.token || !GH_CFG.autoCommit) return; const {owner,repo,branch,token}=GH_CFG; // obtener sha si existe let sha=null; try{ const meta=await fetch(`https://api.github.com/repos/${owner}/${repo}/contents/${path}?ref=${branch}`,{ headers:{"Accept":"application/vnd.github+json","Authorization":`Bearer ${token}`} }); if(meta.status===200){ const j=await meta.json(); sha=j.sha; } }catch{} // PUT await fetch(`https://api.github.com/repos/${owner}/${repo}/contents/${path}`,{ method:"PUT", headers:{ "Accept":"application/vnd.github+json", "Authorization":`Bearer ${token}`, "Content-Type":"application/json" }, body:JSON.stringify({message:msg,branch,content:b64(JSON.stringify(content,null,2)),sha}) }); } // === Render (respeta el orden de las llaves del objeto) === function render(tabla,arr){ if(!arr.length){ tabla.innerHTML="Sin registros"; return; } const cols = Object.keys(arr[0]); // respeta el orden de inserción tabla.innerHTML = ""+cols.map(c=>`${c}`).join("")+""+ ""+ arr.map(r=>""+cols.map(c=>`${r[c]}`).join("")+"").join("")+ ""; }); } function migrarCaex(arr){ return arr.map(r=>{ if("Total Vendido LPS" in r && "Costo Envío" in r && "Cantidad" in r) return r; const nombre = r.Producto || ""; const cant = toNum(r.Cantidad ?? r.Cant); const prod = buscarProd(nombre) || {venta:0, costo:0}; const envio = toNum(r["Costo Envío"] ?? r["Costo Envio"]); const desc = toNum(r.Descuento); const imp = (r.Impuesto || "").toString(); const totalVend = (toNum(prod.venta) * cant); // Caex no suma envío al total vendido const gan = ((toNum(prod.venta)-toNum(prod.costo))*cant) - envio - desc; return { "Fecha": r.Fecha || fechaHora(), "Producto": nombre, "Cantidad": cant, "Costo Envío": envio, "Total Vendido LPS": totalVend.toFixed(2), "Descuento": desc, "Impuesto": imp, "Ganancia LPS": gan.toFixed(2) };
